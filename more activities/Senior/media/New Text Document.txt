    <script>
      // --- DOM Elements ---
      const descriptionContainer = document.getElementById("description-container");
      const feedbackDiv = document.getElementById("feedback");
      const tryAgainBtn = document.getElementById("try-again-btn");
      const checkBtn = document.getElementById("check-btn");
      const canvas = document.getElementById("reward-canvas");
      const ctx = canvas.getContext("2d");
      const magicWand = document.getElementById("magic-wand");

      // --- Game Data ---
      const descriptions = [
        { parts: [ { text: "A boy is smiling and swinging high " }, { options: ["at", "on", "in"], correct: "at" }, { text: " the playground." } ] },
        { parts: [ { text: "Two girls are riding the see-saw and laughing " }, { options: ["together", "apart", "beside"], correct: "together" }, { text: "." } ] },
        { parts: [ { text: "A cute puppy is chasing butterflies " }, { options: ["near", "over", "through"], correct: "near" }, { text: " the flowers." } ] },
        { parts: [ { text: "Parents are sitting " }, { options: ["on", "under", "in"], correct: "on" }, { text: " benches, watching the kids play." } ] },
        { parts: [ { text: "An ice-cream man is giving " }, { options: ["out", "in", "off"], correct: "out" }, { text: " yummy cold treats." } ] },
        { parts: [ { text: "Big trees make nice shady spots " }, { options: ["around", "above", "between"], correct: "around" }, { text: " the park." } ] },
        { parts: [ { text: "A kite is flying " }, { options: ["in", "on", "at"], correct: "in" }, { text: " the sky as a kid runs below." } ] },
        { parts: [ { text: "Two toddlers are having fun building a sandcastle " }, { options: ["with", "by", "for"], correct: "with" }, { text: " sand." } ] }, // Modified for clarity
        { parts: [ { text: "You can hear laughter, birds, and the wind " }, { options: ["in", "at", "on"], correct: "in" }, { text: " the trees." } ] },
        { parts: [ { text: "Everything looks bright and happy " }, { options: ["with", "of", "from"], correct: "with" }, { text: " lots of colors!" } ] },
      ];

      // --- Game Logic ---
      function initializeGame() {
        descriptionContainer.innerHTML = "";
        feedbackDiv.textContent = "";
        feedbackDiv.className = "feedback-container";
        checkBtn.disabled = false;

        descriptions.forEach(desc => {
          const sentenceEl = document.createElement("p");
          sentenceEl.className = "description-sentence";

          desc.parts.forEach(part => {
            if (part.text) {
              sentenceEl.appendChild(document.createTextNode(part.text));
            } else if (part.options) {
              const selectEl = document.createElement("select");
              selectEl.className = "preposition-select";
              selectEl.dataset.correct = part.correct;

              // Add a blank/default option first
              const defaultOption = document.createElement("option");
              defaultOption.value = "";
              defaultOption.textContent = "---";
              selectEl.appendChild(defaultOption);

              const shuffledOptions = [...part.options].sort(() => Math.random() - 0.5);
              shuffledOptions.forEach(optionText => {
                const optionEl = document.createElement("option");
                optionEl.value = optionText;
                optionEl.textContent = optionText;
                selectEl.appendChild(optionEl);
              });
              sentenceEl.appendChild(selectEl);
            }
          });
          descriptionContainer.appendChild(sentenceEl);
        });
      }

      function checkAnswers() {
        const selects = descriptionContainer.querySelectorAll(".preposition-select");
        let allCorrect = true;

        selects.forEach(select => {
          const isCorrect = select.value === select.dataset.correct;
          select.classList.remove("correct", "incorrect");
          select.classList.add(isCorrect ? "correct" : "incorrect");
          if (!isCorrect) {
            allCorrect = false;
          }
        });

        if (allCorrect) {
          feedbackDiv.textContent = "Excellent! You got everything right!";
          feedbackDiv.className = "feedback-container correct-text";
          triggerReward();
          checkBtn.disabled = true;
        } else {
          feedbackDiv.textContent = "Some are incorrect. Please try again!";
          feedbackDiv.className = "feedback-container incorrect-text";
        }
      }

      function resetGame() {
        initializeGame();
      }

      // --- Reward Animation ---
      let animationFrameId;
      function triggerReward() {
        magicWand.style.opacity = "1";
        magicWand.style.transform = "rotate(-45deg) scale(1.2)";
        setTimeout(() => { magicWand.style.transform = "rotate(0deg) scale(1)"; }, 300);
        setTimeout(() => { magicWand.style.opacity = "0"; }, 1500);
        
        const wandRect = magicWand.getBoundingClientRect();
        const startX = wandRect.left - 20;
        const startY = wandRect.top - 20;
        let localStars = [];
        
        for (let i = 0; i < 70; i++) {
          const radius = Math.random() * 3 + 1;
          const color = `hsl(${Math.random() * 60 + 200}, 100%, 70%)`;
          const angle = Math.random() * Math.PI * 0.8 + Math.PI * 1.1;
          const speed = Math.random() * 6 + 4;
          const velocity = { x: Math.cos(angle) * speed, y: Math.sin(angle) * speed };
          localStars.push({ x: startX, y: startY, radius, color, velocity, alpha: 1 });
        }

        function animate() {
          animationFrameId = requestAnimationFrame(animate);
          ctx.clearRect(0, 0, canvas.width, canvas.height);
          localStars.forEach((star, index) => {
            star.x += star.velocity.x;
            star.y += star.velocity.y;
            star.alpha -= 0.015;
            
            ctx.save();
            ctx.globalAlpha = star.alpha;
            ctx.beginPath();
            ctx.arc(star.x, star.y, star.radius, 0, Math.PI * 2, false);
            ctx.fillStyle = star.color;
            ctx.fill();
            ctx.restore();

            if (star.alpha <= 0) localStars.splice(index, 1);
          });
          if (localStars.length === 0) {
            cancelAnimationFrame(animationFrameId);
            animationFrameId = null;
          }
        }
        if (!animationFrameId) {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
            animate();
        }
      }

      // --- Event Listeners ---
      checkBtn.addEventListener("click", checkAnswers);
      tryAgainBtn.addEventListener("click", resetGame);

      // --- Initialization ---
      window.onload = () => {
        initializeGame();
      };
    </script>
